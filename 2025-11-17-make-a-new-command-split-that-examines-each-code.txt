  Helper Class: CellSplitter

  class CellSplitter:
      """Handles the splitting of individual cells using LLM."""

      def __init__(self, model: str):
          self.model = model
          self._setup_agent()

      def _setup_agent(self):
          """Initialize the FerretAgent for cell splitting."""
          self.agent = FerretAgent[SplitCellResponse](
              key="cell_splitting",
              model=self.model,
              instructions=get_prompt("split_instructions"),
              output_type=SplitCellResponse,
          )

      async def split_cell(
          self,
          cell: Dict[str, Any],
          cell_index: int,
          context: str,
      ) -> Tuple[SplitCellResponse, FerretStats]:
          """
          Split a single cell using LLM.
          
          Args:
              cell: The notebook cell to split
              cell_index: Index of the cell in the notebook
              context: Context from previous cells
          
          Returns:
              Tuple of (SplitCellResponse, FerretStats)
          """

          with timer(key=f"split_cell_{cell_index}", message=f"Splitting 
  cell {cell_index}"):
              cell_id = cell.get("id", f"cell_{cell_index}")
              cell_source = "".join(cell.get("source", []))

              # Build input for LLM
              input_text = get_prompt(
                  "split_input",
                  cell_index=cell_index,
                  cell_id=cell_id,
                  cell_source=cell_source,
                  context=context,
              )

              # Call LLM
              log(f"  Analyzing cell {cell_index} ({len(cell_source)} 
  chars)")
              response, stats = await self.agent.run(input_text)

              log(f"  → Should split: {response.should_split}")
              if response.should_split:
                  log(f"  → Splitting into {len(response.split_cells)} 
  cells")

              return response, stats

  Helper Class: ContextBuilder

  class ContextBuilder:
      """Builds context from previous cells for LLM."""

      MAX_CONTEXT_LINES = 50  # Limit context to avoid token explosion

      @staticmethod
      def build_context(cells: List[Dict[str, Any]], current_index: int) -> 
  str:
          """
          Build context string from previous cells.
          
          Args:
              cells: All cells in the notebook
              current_index: Index of current cell being processed
          
          Returns:
              Context string summarizing previous cells
          """

          if current_index == 0:
              return "This is the first cell in the notebook."

          # Collect previous code cells
          prev_cells = []
          for i in range(max(0, current_index - 5), current_index):
              cell = cells[i]
              if cell.get("cell_type") == "code":
                  source = "".join(cell.get("source", []))
                  # Truncate long cells
                  lines = source.split("\n")
                  if len(lines) > 10:
                      source = "\n".join(lines[:10]) + "\n... (truncated)"
                  prev_cells.append(f"Cell {i}:\n{source}")

          if not prev_cells:
              return "No previous code cells."

          return "\n\n".join(prev_cells[-3:])  # Last 3 cells max

  Helper Class: StatsAggregator (similar to optimize.py)

  class StatsAggregator:
      """Aggregates statistics from multiple LLM calls."""

      def __init__(self):
          self.all_stats: List[FerretStats] = []

      def add_stats(self, stats: FerretStats) -> None:
          """Add stats from one LLM call."""
          self.all_stats.append(stats)

      def get_aggregated_stats(self) -> FerretStats:
          """Get aggregated stats across all calls."""
          if not self.all_stats:
              return FerretStats(
                  model="unknown",
                  time=0.0,
                  usage=Usage(input_tokens=0, output_tokens=0),
                  cost=0.0,
                  log_path=None,
              )

          total_time = sum(s.time for s in self.all_stats)
          total_input = sum(s.usage.input_tokens for s in self.all_stats)
          total_output = sum(s.usage.output_tokens for s in self.all_stats)
          total_cost = sum(s.cost for s in self.all_stats)

          return FerretStats(
              model=self.all_stats[0].model,
              time=total_time,
              usage=Usage(input_tokens=total_input,
  output_tokens=total_output),
              cost=total_cost,
              log_path=None,
          )

  Main Processing Logic:

  async def _split_cells(
      self,
      notebook_content: Dict[str, Any],
      selected_cell_ids: Optional[List[str]],
      model: str,
      stats_agg: StatsAggregator,
  ) -> Tuple[Dict[str, Any], Dict[str, Any]]:
      """Process all cells and perform splitting."""

      import copy
      new_notebook = copy.deepcopy(notebook_content)
      cells = new_notebook.get("cells", [])

      # Initialize helpers
      splitter = CellSplitter(model)
      context_builder = ContextBuilder()

      # Track results
      cells_analyzed = 0
      cells_split = 0
      total_new_cells = 0

      # Process cells (in reverse to maintain indices during replacement)
      new_cells_list = []

      with indent(message="Processing cells"):
          for i, cell in enumerate(cells):
              # Skip non-code cells
              if cell.get("cell_type") != "code":
                  new_cells_list.append(cell)
                  continue

              # Skip if not in selected cells (when specified)
              cell_id = cell.get("id")
              if selected_cell_ids and cell_id not in selected_cell_ids:
                  new_cells_list.append(cell)
                  continue

              cells_analyzed += 1

              # Build context
              context = context_builder.build_context(cells, i)

              # Split the cell
              response, stats = await splitter.split_cell(cell, i, context)
              stats_agg.add_stats(stats)

              # Apply splitting if recommended
              if response.should_split:
                  cells_split += 1

                  # Convert split cells to notebook format
                  for split_cell_info in response.split_cells:
                      new_cell = {
                          "cell_type": split_cell_info.cell_type,
                          "metadata": {
                              "split_from": cell_id,
                              "split_description":
  split_cell_info.description,
                          },
                          "source": split_cell_info.source,
                      }

                      # Add execution_count for code cells
                      if split_cell_info.cell_type == "code":
                          new_cell["execution_count"] = None
                          new_cell["outputs"] = []

                      new_cells_list.append(new_cell)
                      total_new_cells += 1

                  log(f"  Split cell {i} into {len(response.split_cells)} 
  cells")
              else:
                  # Keep original cell
                  new_cells_list.append(cell)

      # Update notebook with new cells
      new_notebook["cells"] = new_cells_list

      results = {
          "cells_analyzed": cells_analyzed,
          "cells_split": cells_split,
          "total_new_cells": total_new_cells,
      }

      return new_notebook, results

  4. CLI Tool (data_ferret/cli/split_cli.py)

  """
  Command-line interface for splitting notebook cells.

  Usage:
      data_ferret_split input.ipynb -o output.ipynb
  """

  import asyncio
  import sys
  from pathlib import Path
  from typing import Optional
  import json

  from data_ferret.server.registry import CommandRegistry
  from data_ferret.cli.helpers import load_notebook, save_notebook,
  display_results_table
  from data_ferret.util.output import log, error, timer
  from data_ferret.util.config import load_config


  async def split_notebook(
      input_path: Path,
      output_path: Optional[Path] = None,
  ) -> None:
      """
      Split cells in a notebook.
      
      Args:
          input_path: Path to input notebook
          output_path: Path to save split notebook (default: 
  input_split.ipynb)
      """

      with timer(key="split_cli_total", message="Split notebook CLI"):
          # Load notebook
          log(f"Loading notebook: {input_path}")
          notebook_content = load_notebook(input_path)

          # Load config
          config = load_config()

          # Get split command
          registry = CommandRegistry()
          split_cmd = registry.get_command("split")

          if not split_cmd:
              error("Split command not found in registry")
              sys.exit(1)

          # Execute split
          log("Executing split command...")
          result = await split_cmd.process(
              notebook_content=notebook_content,
              kernel_client=None,  # Not needed for split
              config=config,
          )

          # Extract results
          split_notebook = result["notebook"]
          metadata = result["metadata"]

          # Determine output path
          if output_path is None:
              output_path = input_path.parent /
  f"{input_path.stem}_split.ipynb"

          # Save split notebook
          log(f"Saving split notebook: {output_path}")
          save_notebook(split_notebook, output_path)

          # Display results
          log("\n" + "="*60)
          log("SPLIT RESULTS")
          log("="*60)
          log(f"Cells analyzed:  {metadata['cells_analyzed']}")
          log(f"Cells split:     {metadata['cells_split']}")
          log(f"New cells added: {metadata['total_new_cells']}")
          log(f"\nLLM Usage:")
          llm_stats = metadata['llm_stats']
          log(f"  Model:         {llm_stats['model']}")
          log(f"  Cost:          ${llm_stats['cost']:.4f}")
          log(f"  Time:          {llm_stats['time']:.2f}s")
          log(f"  Input tokens:  {llm_stats['input_tokens']:,}")
          log(f"  Output tokens: {llm_stats['output_tokens']:,}")
          log("="*60)

          log(f"\nSplit notebook saved to: {output_path}")


  def split_cli_main():
      """Entry point for data_ferret_split CLI."""
      import argparse

      parser = argparse.ArgumentParser(
          description="Split Jupyter notebook cells into logical steps"
      )
      parser.add_argument(
          "input",
          type=Path,
          help="Input notebook path (.ipynb)",
      )
      parser.add_argument(
          "-o", "--output",
          type=Path,
          help="Output notebook path (default: input_split.ipynb)",
      )

      args = parser.parse_args()

      # Validate input
      if not args.input.exists():
          error(f"Input notebook not found: {args.input}")
          sys.exit(1)

      if not args.input.suffix == ".ipynb":
          error(f"Input must be a .ipynb file: {args.input}")
          sys.exit(1)

      # Run async split
      asyncio.run(split_notebook(args.input, args.output))


  if __name__ == "__main__":
      split_cli_main()
